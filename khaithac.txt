function formatDateForInput(iso) {
  if(!iso) return "";
  const d = new Date(iso);
  return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
}

function formatTimeForInput(val){
  if(!val) return "";

  // ✅ Nếu đã là dạng HH:mm thì giữ nguyên (QUAN TRỌNG)
  if (typeof val === "string" && /^\d{2}:\d{2}$/.test(val)) {
    return val;
  }

  // ✅ Nếu là ISO (có T hoặc Z) thì mới parse Date
  if (typeof val === "string" && val.includes("T")) {
    const d = new Date(val);
    return `${String(d.getHours()).padStart(2,'0')}:${String(d.getMinutes()).padStart(2,'0')}`;
  }

  return "";
}

function applyBlinkEffect(cellElement, planned, actual){
  cellElement.classList.remove("blink-red","blink-green","solid-green","blink-gold");
  if(planned && actual){
    if(actual > planned) cellElement.classList.add("blink-red");
    else cellElement.classList.add("solid-green");
  } else {
    cellElement.classList.add("blink-gold");
  }
}

function parseTime(str){
  if(!str) return null;
  const [h,m] = str.split(":").map(Number);
  return h*60 + m; // tổng số phút
}

function formatMinutes(total){
  if(total==null) return "";
  const h = String(Math.floor(total/60)).padStart(2,"0");
  const m = String(total%60).padStart(2,"0");
  return `${h}:${m}`;
}

/// giờ utc
function convertLocalMinus7(timeStr, dateStr){
  if(!timeStr || !dateStr) return {time:"", day:""};

  // timeStr dạng HH:MM
  const [hh, mm] = timeStr.split(":").map(Number);
  const [year, month, day] = dateStr.split("-").map(Number);

  // Tạo đối tượng Date theo giờ local
  const localDate = new Date(year, month-1, day, hh, mm);

  // Trừ đi 7 tiếng
  const newDate = new Date(localDate.getTime() - 7*60*60*1000);

  // Lấy giờ và phút
  const hours = newDate.getHours().toString().padStart(2,"0");
  const minutes = newDate.getMinutes().toString().padStart(2,"0");

  // Bỏ dấu “:”
  const timeOut = hours + minutes;

  // Lấy ngày mới (dd)
  const newDay = newDate.getDate().toString().padStart(2,"0");

  return {time: timeOut, day: newDay};

function normalizeTime(t){
  if(!t) return "";
  if(t.includes(":")) return t;
  return t.slice(0,2) + ":" + t.slice(2);
}

// Hàm cộng phút vào chuỗi giờ "HH:MM"
function addMinutesToTime(timeStr, minutes){
  const [hh, mm] = timeStr.split(":").map(Number);
  if(isNaN(hh) || isNaN(mm)) return "";

  const d = new Date();
  d.setHours(hh);
  d.setMinutes(mm + minutes);

  const newH = String(d.getHours()).padStart(2,"0");
  const newM = String(d.getMinutes()).padStart(2,"0");
  return newH + newM; // xuất dạng HHMM
}

  let delayMinutes = null;
  if(fhtTime && etdTime){
    delayMinutes = parseTime(fhtTime) - parseTime(etdTime);
    delay = (delayMinutes >= 0 ? delayMinutes : delayMinutes) + " Phút";
  }

  // ✅❌ LOGIC PASS / FAIL
  let status = "⛔FAIL";

  if(fhtTime && etdTime && ataTime){
    const fhtMin = parseTime(fhtTime);
    const etdMin = parseTime(etdTime);
    const ataMin = parseTime(ataTime);

    if(fhtMin < etdMin || (fhtMin - ataMin) <= 35){
      status = "✅PASS";
    }
  }

  const text = `${depFlight} ${dateFormatted}
C.ON ${ataTime.replace(":","")} FHT ${fhtTime.replace(":","")}
GT ${gt}
FHT SO ETD ${delay}
${status}`;

  document.getElementById("fhtOutput").textContent = text;

  // clear output khác
  document.getElementById("mvaOutput").textContent = "";
  document.getElementById("mvdOutput").textContent = "";
  document.getElementById("occOutput").textContent = "";
}

function showOCC(row){
  if(!row){
    alert("Bạn chưa chọn dòng (double click vào bảng)");
    return;
  }

  const dateStr = formatDateForInput(row[1]);
  const depFlight = (row[9] || "").toUpperCase();
  const from = (row[4] || "").toUpperCase();
  const to = (row[10] || "").toUpperCase();
  const acReg = (row[2] || "").toUpperCase();

  const std = formatTimeForInput(row[11]);
  const pushback = formatTimeForInput(row[15]);

  const code1 = (row[19] || "").toUpperCase();
  const time1 = (row[20] || "").toUpperCase();
  const code2 = (row[21] || "").toUpperCase();
  const time2 = (row[22] || "").toUpperCase();
  const config = (row[25] || "").toUpperCase();
  const details = (row[28] || "").toUpperCase();

  // ===== MAP DELAY CAUSE =====
  const delayCauseMap = {
    OA:"KHÔNG CÓ GATE/ BÃI ĐỖ", SG:"GROUNDTIME THẤP HƠN TIÊU CHUẨN",
    PD:"NHẬN KHÁCH MUỘN", PL:"QUẦY THỦ TỤC QUÁ TẢI",
    PE:"LỖI CHỦ QUAN DO LÀM THỦ TỤC", PO:"OVERBOOK, LỖI BOOKING",
    PH:"LỖI BOARDING", PS:"THIẾU TIỆN NGHI PHỤC VỤ KHÁCH/ KHÁCH BỊ MẤT ĐỒ",
    PC:"ĐẶT SUẤT ĂN SAI", PB:"PHÂN LOẠI HÀNH LÝ CHẬM",
    PW:"KHÁCH HẠN CHẾ DI CHUYỂN",

    CD:"SAI TÀI LIỆU HÀNG HÓA", CP:"HÀNG HÓA TẬP KẾT CHẬM",
    CC:"CHẤP NHẬN HÀNG HÓA CHẬM", CI:"LỖI ĐÓNG GÓI HÀNG HÓA",
    CO:"LỖI ĐẶT CHỖ / BÁN QUÁ CHỖ", CU:"CARGO KÉO RA TÀU MUỘN",

    GD:"TÀI LIỆU CHUYẾN BAY SAI, CHẬM", GE:"THIẾU PHƯƠNG TIỆN CHẤT XẾP",
    GS:"THIẾU PHƯƠNG TIỆN PHỤC VỤ KHÁCH", GC:"VỆ SINH TÀU BAY CHẬM",
    GF:"NẠP/ RÚT DẦU",

    TD:"LỖI KỸ THUẬT", TM:"BẢO DƯỠNG CHẬM",
    TN:"BẢO DƯỠNG BẤT THƯỜNG", TC:"ĐỔI TÀU BAY DO KỸ THUẬT",
    TL:"THIẾU TÀU BAY DỰ BỊ", TV:"THAY ĐỔI CONFIG",

    WO:"THỜI TIẾT TẠI SÂN BAY KHỞI HÀNH",
    WT:"THỜI TIẾT TẠI SÂN BAY ĐẾN",
    WI:"CHỐNG PHÁ BĂNG",

    AD:"ATC", AM:"ATC",

    RA:"TÀU VỀ MUỘN", RO:"QUẢN LÝ ĐIỀU HÀNH KHAI THÁC",
    MX:"BÁO CÁO TRONG DETAILS", AF:"QUÁ NĂNG LỰC CHK"
  };

  function getCause(code){
    return delayCauseMap[(code || "").trim()] || "UNKNOWN";
  }

  function toMinutes(t){
    if(!t) return 0;
    return parseInt(t,10) || 0;
  }

  // ===== FORMAT DATE =====
  let dateFormatted = "";
  if(dateStr){
    const [y,m,d] = dateStr.split("-");
    dateFormatted = `${d}/${m}/${y}`;
  }

  // ===== FORMAT TIME =====
  const stdText = std ? std.replace(":","") : "";
  const pushbackText = pushback ? pushback.replace(":","") : "";

  // ===== TÍNH DELAY =====
  let delayMinutes = 0;
  if(std && pushback){
    delayMinutes = parseTime(pushback) - parseTime(std);
    if(delayMinutes < 0) delayMinutes = 0;
  }

  let delayText = delayMinutes ? delayMinutes + " phút" : "";

  // ===== DELAY CODE =====
  let delayCode = "";
  let causeText = "";

  if(code1 && !code2){
    delayCode = `${code1}${time1}`;
    causeText = getCause(code1);
  }
  else if(code1 && code2){
    delayCode = `${code1}${time1} + ${code2}${time2}`;
    causeText = `${getCause(code1)} + ${getCause(code2)}`;
  }

  // ===== CHECK THIẾU DELAY → ADD AM =====
  const totalCodeTime = toMinutes(time1) + toMinutes(time2);

  if(delayMinutes > totalCodeTime && delayCode){
    const extra = delayMinutes - totalCodeTime;
    delayCode += ` + AM${extra}`;
    causeText += ` + ATC`;
  }

  // ===== OUTPUT =====
  const text = `Station : PQC
Date: ${dateFormatted}
Reporter : VNPQC 
Flight : ${depFlight}
Route : PQC${to}
A/C Reg : ${acReg}
A/C Type : ${config}
STD : ${stdText}
ATD : ${pushbackText}
Delay : ${delayText}
Delay code : ${delayCode}
Delay Cause : ${causeText}
Details :${details}`;

  document.getElementById("occOutput").textContent = text;

  // clear output khác
  document.getElementById("mvaOutput").textContent = "";
  document.getElementById("mvdOutput").textContent = "";
  document.getElementById("fhtOutput").textContent = "";
}
