function formatDateForInput(iso) {
  if(!iso) return "";
  const d = new Date(iso);
  return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
}

function formatTimeForInput(val){
  if(!val) return "";

  // ✅ Nếu đã là dạng HH:mm thì giữ nguyên (QUAN TRỌNG)
  if (typeof val === "string" && /^\d{2}:\d{2}$/.test(val)) {
    return val;
  }

  // ✅ Nếu là ISO (có T hoặc Z) thì mới parse Date
  if (typeof val === "string" && val.includes("T")) {
    const d = new Date(val);
    return `${String(d.getHours()).padStart(2,'0')}:${String(d.getMinutes()).padStart(2,'0')}`;
  }

  return "";
}
 function autoFormatTime(input) {
    let value = input.value.replace(/[^0-9]/g, ""); // chỉ giữ số

    if (value.length === 4) {
      let hours = parseInt(value.slice(0, 2), 10);
      let minutes = parseInt(value.slice(2, 4), 10);

      // Giới hạn giờ và phút
      if (hours > 23) hours = 23;
      if (minutes > 59) minutes = 59;

      // Đảm bảo luôn có 2 chữ số
      let hh = hours.toString().padStart(2, "0");
      let mm = minutes.toString().padStart(2, "0");

      input.value = hh + ":" + mm;
    } else {
      input.value = value;
    }
  }

  // Gắn sự kiện cho tất cả ô nhập giờ
  document.querySelectorAll('#sta, #eta,#landing, #ata, #std, #etd, #atd, #fht, #pushback,#takeoff, #delay')
    .forEach(el => {
      el.addEventListener('input', () => autoFormatTime(el));
    });

function applyBlinkEffect(cellElement, planned, actual){
  cellElement.classList.remove("blink-red","blink-green","solid-green","blink-gold");
  if(planned && actual){
    if(actual > planned) cellElement.classList.add("blink-red");
    else cellElement.classList.add("solid-green");
  } else {
    cellElement.classList.add("blink-gold");
  }
}

function parseTime(str){
  if(!str) return null;
  const [h,m] = str.split(":").map(Number);
  return h*60 + m; // tổng số phút
}

function formatMinutes(total){
  if(total==null) return "";
  const h = String(Math.floor(total/60)).padStart(2,"0");
  const m = String(total%60).padStart(2,"0");
  return `${h}:${m}`;
}

function convertLocalMinus7(timeStr, dateStr){
  if(!timeStr || !dateStr) return {time:"", day:""};

  // timeStr dạng HH:MM
  const [hh, mm] = timeStr.split(":").map(Number);
  const [year, month, day] = dateStr.split("-").map(Number);

  // Tạo đối tượng Date theo giờ local
  const localDate = new Date(year, month-1, day, hh, mm);

  // Trừ đi 7 tiếng
  const newDate = new Date(localDate.getTime() - 7*60*60*1000);

  // Lấy giờ và phút
  const hours = newDate.getHours().toString().padStart(2,"0");
  const minutes = newDate.getMinutes().toString().padStart(2,"0");

  // Bỏ dấu “:”
  const timeOut = hours + minutes;

  // Lấy ngày mới (dd)
  const newDay = newDate.getDate().toString().padStart(2,"0");

  return {time: timeOut, day: newDay};
}

  const day = landingObj.day || (dateStr ? dateStr.split("-")[2] : "");

  const text = `MVT
${arrFlight}/${day}.${acReg}.PQC
AA${landingObj.time}/${ataObj.time}`;
  document.getElementById("mvaOutput").textContent = "";
  document.getElementById("mvdOutput").textContent = "";

/// dieu chinh lai format gio
function normalizeTime(t){
  if(!t) return "";
  if(t.includes(":")) return t;
  return t.slice(0,2) + ":" + t.slice(2);
}

// Bảng mapping offset phút theo sân bay
const offsetMap = {
  "SGN": 65,   // +1h05
  "HAN": 125,  // +2h05
  "DAD": 45,   // +0h45
  "PQC": 30    // +0h30
};

function showMVD(row){
  if(!row) return;

  const depFlight = (row[9]||"").toUpperCase();
  const dateStr   = formatDateForInput(row[1]);
  const acReg     = (row[2]||"").toUpperCase();
  const to        = (row[10]||"").toUpperCase();

  const adObj    = convertLocalMinus7(formatTimeForInput(row[13]), dateStr); // ATD
  const fht      = convertLocalMinus7(formatTimeForInput(row[14]), dateStr); // FHT
  const takeoff  = convertLocalMinus7(formatTimeForInput(row[16]), dateStr); // ETA

  // 🔹 EA = ETA + offset
let eaTime = takeoff.time;

if(offsetMap[to] && takeoff.time){
  const norm = normalizeTime(takeoff.time);
  eaTime = addMinutesToTime(norm, offsetMap[to]);
}

  const day = adObj.day || (dateStr ? dateStr.split("-")[2] : "");
 const delayCode = (row[19]||"").toUpperCase(); // ví dụ cột 19 chứa delay code

  let text = `MVT
${depFlight}/${day}.${acReg}.PQC
AD${adObj.time}/${takeoff.time}  EA${eaTime}  ${to}`;

  // nếu có delay code thì chèn ngay sau dòng AD
  if(delayCode){
    text += `\nDL${delayCode}`;
  }

  text += `
PX
SI TOF    APLD
FHT ${fht.time}`;

  document.getElementById("mvdOutput").textContent = text;
  document.getElementById("mvaOutput").textContent = "";
}

// Hàm cộng phút vào chuỗi giờ "HH:MM"
function addMinutesToTime(timeStr, minutes){
  const [hh, mm] = timeStr.split(":").map(Number);
  if(isNaN(hh) || isNaN(mm)) return "";

  const d = new Date();
  d.setHours(hh);
  d.setMinutes(mm + minutes);

  const newH = String(d.getHours()).padStart(2,"0");
  const newM = String(d.getMinutes()).padStart(2,"0");
  return newH + newM; // xuất dạng HHMM
}
